🌎 Administração de Usuários
📋 Configuração inicial — clique para ver o SQL necessário

Execute este SQL no Supabase → SQL Editor uma única vez para criar a tabela de perfis:

-- [v48] Tabela de perfis de usuário (plantas)
CREATE TABLE IF NOT EXISTS user_profiles (
  id           UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  email        TEXT UNIQUE NOT NULL,
  unidade      TEXT NOT NULL,
  display_name TEXT,
  updated_at   TIMESTAMPTZ DEFAULT now()
);

-- Habilitar Row Level Security
ALTER TABLE user_profiles ENABLE ROW LEVEL SECURITY;

-- Remove política anterior (evita erro de duplicata) e recria
DROP POLICY IF EXISTS "auth_manage_profiles" ON user_profiles;
DROP POLICY IF EXISTS "allow_authenticated_all" ON user_profiles;

-- Política: qualquer usuário autenticado pode ler e gravar perfis
CREATE POLICY "allow_authenticated_all"
ON user_profiles FOR ALL
TO authenticated
USING (true)
WITH CHECK (true);

-- IMPORTANTE: conceder permissão de tabela ao role authenticated
-- (RLS sozinho não basta — o role precisa ter acesso à tabela)
GRANT ALL ON TABLE user_profiles TO authenticated;
GRANT ALL ON TABLE user_profiles TO service_role;

-- Garantir que a coluna unidade exista na tabela equipamentos
ALTER TABLE equipamentos ADD COLUMN IF NOT EXISTS unidade TEXT;
